Solution 2: Comprehensive Firebase Security Rules

Industry Standard: Defense in Depth Security

Implementing multi-layered security with proper authentication, authorization, and data validation following OWASP guidelines. 

// Firestore Security Rules (firestore.rules)
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // User profiles - users can only access their own data
    match /users/{userId} {
      allow read, write: if request.auth != null 
        && request.auth.uid == userId
        && isValidUserData(request.resource.data);
    }

    // Task completions - users can create, read their own, creators can verify
    match /taskCompletions/{completionId} {
      allow create: if request.auth != null 
        && request.auth.uid == resource.data.userId
        && isValidTaskCompletion(resource.data);
      
      allow read: if request.auth != null 
        && (request.auth.uid == resource.data.userId 
            || request.auth.uid == resource.data.missionCreatorId);
      
      allow update: if request.auth != null 
        && (request.auth.uid == resource.data.userId 
            || request.auth.uid == resource.data.missionCreatorId)
        && isValidTaskCompletionUpdate(request.resource.data);
    }

    // Review assignments - reviewers can read and update
    match /reviewAssignments/{assignmentId} {
      allow read, update: if request.auth != null 
        && request.auth.uid == resource.data.reviewerId
        && isValidReviewAssignment(request.resource.data);
    }

    // Helper functions for data validation
    function isValidUserData(data) {
      return data.keys().hasAll(['email', 'name', 'createdAt'])
        && data.email is string
        && data.name is string
        && data.createdAt is timestamp
        && (data.twitterUsername == null || data.twitterUsername is string);
    }

    function isValidTaskCompletion(data) {
      return data.keys().hasAll(['taskId', 'missionId', 'userId', 'status', 'createdAt'])
        && data.taskId is string
        && data.missionId is string
        && data.userId is string
        && data.status in ['completed', 'verified', 'flagged']
        && data.createdAt is timestamp;
    }

    function isValidTaskCompletionUpdate(data) {
      return data.status in ['completed', 'verified', 'flagged']
        && (data.status == 'flagged' ? data.flagReason is string : true);
    }

    function isValidReviewAssignment(data) {
      return data.keys().hasAll(['submissionId', 'reviewerId', 'status'])
        && data.status in ['pending', 'completed', 'skipped']
        && data.reviewerId is string;
    }
  }
}

// Migration Script (TypeScript)
import { Firestore } from '@google-cloud/firestore';
import { logger } from './utils/logger';
import { validateDataIntegrity } from './utils/validation';

interface MigrationConfig {
  batchSize: number;
  maxRetries: number;
  rollbackEnabled: boolean;
  dryRun: boolean;
}

class DataMigrationService {
  private db: Firestore;
  private config: MigrationConfig;

  constructor(config: MigrationConfig) {
    this.db = new Firestore();
    this.config = config;
  }

  async migrateTaskCompletions(): Promise {
    const startTime = Date.now();
    let processedCount = 0;
    let errorCount = 0;
    const errors: MigrationError[] = [];

    try {
      // Step 1: Backup existing data
      await this.createBackup('mission_participations');
      
      // Step 2: Migrate data in batches
      const batch = this.db.batch();
      const oldCompletions = await this.db
        .collection('mission_participations')
        .limit(this.config.batchSize)
        .get();

      for (const doc of oldCompletions.docs) {
        try {
          const newData = this.transformTaskCompletion(doc.data());
          
          // Validate data integrity
          if (!validateDataIntegrity(newData)) {
            throw new Error(`Invalid data for document ${doc.id}`);
          }

          // Create new document
          const newDocRef = this.db.collection('taskCompletions').doc();
          batch.set(newDocRef, {
            ...newData,
            migratedFrom: doc.id,
            migratedAt: new Date(),
            migrationVersion: '1.0.0'
          });

          processedCount++;
        } catch (error) {
          errorCount++;
          errors.push({
            documentId: doc.id,
            error: error.message,
            timestamp: new Date()
          });
        }
      }

      // Step 3: Commit batch if not dry run
      if (!this.config.dryRun) {
        await batch.commit();
      }

      // Step 4: Validate migration
      await this.validateMigration(processedCount);

      return {
        success: errorCount === 0,
        processedCount,
        errorCount,
        errors,
        duration: Date.now() - startTime
      };

    } catch (error) {
      logger.error('Migration failed', error);
      
      // Rollback if enabled
      if (this.config.rollbackEnabled) {
        await this.rollback();
      }
      
      throw error;
    }
  }

Industry Standard: Circuit Breaker Pattern with Retry Logic

Implementing comprehensive error handling with circuit breakers, exponential backoff, and graceful degradation following enterprise patterns. 

// Error Handling Service (TypeScript)
import { CircuitBreaker } from 'opossum';
import { logger } from './utils/logger';
import { Sentry } from '@sentry/nextjs';

export enum ErrorType {
  NETWORK_ERROR = 'NETWORK_ERROR',
  VALIDATION_ERROR = 'VALIDATION_ERROR',
  AUTHENTICATION_ERROR = 'AUTHENTICATION_ERROR',
  RATE_LIMIT_ERROR = 'RATE_LIMIT_ERROR',
  FIREBASE_ERROR = 'FIREBASE_ERROR',
  UNKNOWN_ERROR = 'UNKNOWN_ERROR'
}

export interface ErrorContext {
  userId?: string;
  operation: string;
  timestamp: Date;
  requestId: string;
  metadata?: Record;
}

export class ErrorHandlingService {
  private circuitBreakers: Map = new Map();

  constructor() {
    this.initializeCircuitBreakers();
  }

  private initializeCircuitBreakers(): void {
    // Firebase operations circuit breaker
    this.circuitBreakers.set('firebase', new CircuitBreaker(
      this.firebaseOperation.bind(this),
      {
        timeout: 5000,
        errorThresholdPercentage: 50,
        resetTimeout: 30000,
        rollingCountTimeout: 10000,
        rollingCountBuckets: 10
      }
    ));

    // API operations circuit breaker
    this.circuitBreakers.set('api', new CircuitBreaker(
      this.apiOperation.bind(this),
      {
        timeout: 10000,
        errorThresholdPercentage: 60,
        resetTimeout: 60000,
        rollingCountTimeout: 15000,
        rollingCountBuckets: 10
      }
    ));
  }

  async handleError(
    error: Error, 
    context: ErrorContext, 
    retryable: boolean = true
  ): Promise {
    const errorType = this.classifyError(error);
    const errorId = this.generateErrorId();

    // Log error with context
    logger.error('Application error occurred', {
      errorId,
      errorType,
      message: error.message,
      stack: error.stack,
      context
    });

    // Send to monitoring service
    Sentry.captureException(error, {
      tags: {
        errorType,
        operation: context.operation,
        userId: context.userId
      },
      extra: context.metadata
    });

    // Handle specific error types
    switch (errorType) {
      case ErrorType.NETWORK_ERROR:
        return this.handleNetworkError(error, context, retryable);
      case ErrorType.RATE_LIMIT_ERROR:
        return this.handleRateLimitError(error, context);
      case ErrorType.AUTHENTICATION_ERROR:
        return this.handleAuthenticationError(error, context);
      case ErrorType.VALIDATION_ERROR:
        return this.handleValidationError(error, context);
      default:
        return this.handleUnknownError(error, context);
    }
  }

  private classifyError(error: Error): ErrorType {
    if (error.message.includes('network') || error.message.includes('fetch')) {
      return ErrorType.NETWORK_ERROR;
    }
    if (error.message.includes('rate limit') || error.message.includes('429')) {
      return ErrorType.RATE_LIMIT_ERROR;
    }
    if (error.message.includes('auth') || error.message.includes('401')) {
      return ErrorType.AUTHENTICATION_ERROR;
    }
    if (error.message.includes('validation') || error.message.includes('400')) {
      return ErrorType.VALIDATION_ERROR;
    }
    if (error.message.includes('firebase') || error.message.includes('firestore')) {
      return ErrorType.FIREBASE_ERROR;
    }
    return ErrorType.UNKNOWN_ERROR;
  }

  private async handleNetworkError(
    error: Error, 
    context: ErrorContext, 
    retryable: boolean
  ): Promise {
    if (retryable) {
      // Implement exponential backoff retry
      await this.retryWithBackoff(context.operation, 3);
    }
    
    throw new UserFriendlyError(
      'Network connection issue. Please check your internet connection and try again.',
      'NETWORK_ERROR',
      context
    );
  }

  private async retryWithBackoff(operation: string, maxRetries: number): Promise {
    for (let attempt = 1; attempt <= maxRetries; attempt++) {
      const delay = Math.pow(2, attempt) * 1000; // Exponential backoff
      await new Promise(resolve => setTimeout(resolve, delay));
      
      try {
        // Retry the operation
        return;
      } catch (error) {
        if (attempt === maxRetries) {
          throw error;
        }
      }
    }
  }
}

// Custom error classes
export class UserFriendlyError extends Error {
  constructor(
    message: string,
    public code: string,
    public context: ErrorContext
  ) {
    super(message);
    this.name = 'UserFriendlyError';
  }
}
